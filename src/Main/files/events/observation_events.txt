country_event = {
	id = observation.2
	title = "observation.2.name"
	desc = "observation.2.desc"
	picture = GFX_evt_asteroid_approaching_planet
	show_sound = event_red_alert
	location = event_target:primitive_impact_planet

	is_triggered_only = yes
	
	immediate = {
		create_country = {
			name = "NAME_Incoming_Asteroid"
			type = faction
		}
		last_created_country = {
			set_country_flag = incoming_asteroid_prim
			create_fleet = {
				name = "NAME_Asteroid"
				effect = {
					set_owner = PREV
					set_fleet_flag = primitive_asteroid
					create_ship = {
						name = "NAME_Asteroid"
						design = "NAME_Asteroid"
					}
					set_location = {
						target = FROM
						distance = 75
						angle = random
					}
					queue_actions = {
						move_to = FROM
						effect = {
							id = "observation.2.effect.1"
							ROOT = {
								random_list = {
									75 = { country_event = { id = observation.3 } }
									25 = {
										country_event = { id = observation.7 }
										modifier = {
											factor = 0.1
											has_ethic = ethic_xenophobe
										}
										modifier = {
											factor = 0
											has_ethic = ethic_fanatic_xenophobe
										}
									}
								}
							}
							destroy_fleet = this
						}
					}
				}
			}
		}
	}
	
	option = {
		name = "observation.2.a"
	}
}

		shift_ethic = "ethic_authoritarian"

		}
	}

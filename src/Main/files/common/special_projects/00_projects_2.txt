	key = MANUFACTORY_PROJECT
	days_to_research = 80

	requirements = {
		shipclass_military = 2
	}

	abort_trigger = {
		NOT = {
			fromfrom = {
				solar_system = {
					has_access_fleet = root

		if = {
			limit = {

					has_country_flag = manufactory4

			}
			ship_event = { id = anomaly.3126 }
		}

			limit = {
				owner = {
					has_country_flag = manufactory4

				}
			}

				50 = { ship_event = { id = anomaly.3126 } }

# Possible Variables for Buildings:
# strategic_resource: strategic resource, deprecated, strategic resource used by the building.
# allow: trigger to check for allowing construction of building.
# prerequisites: Tech requirements for building.
# empire_unique: boolean, can only build one if set to true.
# cost: resource table, cost of building.
# is_orbital: boolean, can only be built in orbital station.
# modifier: modifier, deprecated, applies a modifier to planet.
# base_buildtime: int, number of days for construction.
# requires_pop, boolean, building will require a pop for production.
# required_resources, resource table, required resources for production.
# produced_resources, resource table, produced resources in production.
# upgrades, buildings list, buildings this building can upgrade into.
# is_listed, boolean, toggles if this building is shown in the non-upgrade buildable list.
# planet_unique, toggles if one can build multiple of this type on a single planet.
# ai_weight, weight for AI, default is set to one, weight set to 0 means that AI will never build it

#

building_guardian_dragon_shrine = {
	base_buildtime = 360

	empire_unique = yes

	requires_pop = yes
	
	cost = {
		minerals = 800
	}
	
	produced_resources = {
		unity = 15
	}

	potential = {
		planet.owner = {
			has_country_flag = dragon_slayer
			OR = {
				has_ethic = ethic_spiritualist
				has_ethic = ethic_fanatic_spiritualist
				has_ethic = ethic_xenophile
				has_ethic = ethic_fanatic_xenophile
			}
		}
	}
	
	ai_weight = {
		weight = 20
		modifier = {
			weight = 0
			from = { sector_controlled = yes }
			from = { NOT = { has_building = building_guardian_dragon_shrine } }
		}		
	}
	
	ai_replace = no
	event_building = yes
}

# Artist Monument
building_artist = {
	icon = "building_zanaam_obelisk"
	base_buildtime = 400
	planet_unique = yes
	
	produced_resources = {
		unity = 8
	}

	planet_modifier = {
		planet_migration_all_pull = 0.50		
	}
	
	allow = {
		custom_tooltip = {
			text = building_artist_tooltip
			planet = {
				owner = {
					OR = {
						AND = {
							check_variable = {
								which = artist_monument_purchased
								value = 5
							}
							OR = {
								check_variable = {
									which = artist_monuments_built
									value = 4
								}
								check_variable = {
									which = artist_monuments_built
									value = 3
								}
								check_variable = {
									which = artist_monuments_built
									value = 2
								}
								check_variable = {
									which = artist_monuments_built
									value = 1
								}
								check_variable = {
									which = artist_monuments_built
									value = 0
								}
							}
						}AND = {
							check_variable = {
								which = artist_monument_purchased
								value = 4
							}
							OR = {
								check_variable = {
									which = artist_monuments_built
									value = 3
								}
								check_variable = {
									which = artist_monuments_built
									value = 2
								}
								check_variable = {
									which = artist_monuments_built
									value = 1
								}
								check_variable = {
									which = artist_monuments_built
									value = 0
								}
							}
						}
						AND = {
							check_variable = {
								which = artist_monument_purchased
								value = 3
							}
							OR = {
								check_variable = {
									which = artist_monuments_built
									value = 2
								}
								check_variable = {
									which = artist_monuments_built
									value = 1
								}
								check_variable = {
									which = artist_monuments_built
									value = 0
								}
							}
						}
						AND = {
							check_variable = {
								which = artist_monument_purchased
								value = 2
							}
							OR = {
								check_variable = {
									which = artist_monuments_built
									value = 1
								}
								check_variable = {
									which = artist_monuments_built
									value = 0
								}
							}
						}
						AND = {
							check_variable = {
								which = artist_monument_purchased
								value = 1
							}
							check_variable = {
								which = artist_monuments_built
								value = 0
							}
						}
					}
				}
			}
		}
	}

	potential = {
		planet = {
			owner = {
				check_variable = {
					which = artist_monument_purchased
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		planet = {
			owner = {
				if = {
					limit = {
						check_variable = {
							which = artist_monuments_built
							value < 1
						}
					}
					set_variable = {
						which = artist_monuments_built
						value = 1
					}
					else = {
						change_variable = {
							which = artist_monuments_built
							value = 1
						}
					}
				}
			}
		}
	}
	
	on_unqueued = {
		planet = {
			owner = {
				change_variable = {
					which = artist_monuments_built
					value = -1
				}
			}
		}
	}

	ai_weight = {
		weight = 20
		modifier = {
			weight = 0
			from = { sector_controlled = yes }
			from = { NOT = { has_building = building_artist } }
		}		
	}
	
	ai_replace = no
	event_building = yes
}
